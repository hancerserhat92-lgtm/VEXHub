-- Kopyalama fonksiyonu (evrensel)
local function copyLink()
    local success = false
    if setclipboard then
        success = pcall(setclipboard, myLink)
    elseif game:GetService("ClipboardService") then
        success = pcall(function() game:GetService("ClipboardService"):SetText(myLink) end)
    end
    return success
end

    pcall(function() Instance.new("UICorner").Parent = notif end)

    txt.BackgroundTransparency = 1

end

    local running = true
    while running and frame and frame.Parent do
        for hue = 0, 1, 0.02 do
            if not frame or not frame.Parent then running = false break end
            pcall(function()

                frame.BorderColor3 = color

-- Sürükleme (basit ve sağlam)
local dragging = false
local dragStart, frameStart
frame.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 then
        dragging = true
        dragStart = input.Position
        frameStart = frame.Position
        input.Changed:Connect(function()
            if input.UserInputState == Enum.UserInputState.End then dragging = false end
        end)
    end
end)
game:GetService("UserInputService").InputChanged:Connect(function(input)
    if dragging and input.UserInputType == Enum.UserInputType.MouseMovement then
        local delta = input.Position - dragStart
        frame.Position = UDim2.new(frameStart.X.Scale, frameStart.X.Offset + delta.X, frameStart.Y.Scale, frameStart.Y.Offset + delta.Y)
    end
end)
